cmp

Syntax:
cmp <a>, <b>
Functionality:
Compares the values of <a> and <b> and sets EFLAGS
Sample Code

mov     rax, 0x100
mov     rbx, 0x200
cmp     rax, rbx

jmp (cond)

Syntax:
jcc <dst>
Functionality:
Does a conditional jump to <dst> based on EFLAGS
Sample Code

mov     rax, 0x100
mov     rbx, 0x200
cmp     rax, rbx
mov     rcx, rax
jl      rbx_lesser
mov     rcx, rbx
rbx_lesser:
nop
